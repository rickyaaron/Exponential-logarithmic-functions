# Exponential-Logarithmic-Functions

import numpy as np

arr = np.array([3, 4, 5])

print("Exponential (e^x):", np.exp(arr))

print("Natural Log (ln x):", np.log(arr))

print("Log base 10:", np.log10(arr))
